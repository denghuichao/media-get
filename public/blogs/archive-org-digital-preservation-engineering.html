        entry['thumbnails'].append({

            'width': int_or_none(f.get('width')),

            'filesize': int_or_none(f.get('size'))
        })

    # Media file processing
    _, has_ext, extension = f['name'].rpartition('.')
    if not has_ext:

    # Privacy-aware format filtering

            'format': f.get('format'),
